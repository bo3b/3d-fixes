void main(
  float4 injectedScreenPos : SV_Position,
  float4 v0 : SV_Position0,
  float4 v1 : TEXCOORD0,
  float3 v2 : TEXCOORD1,
  float3 v3 : TEXCOORD2,
  out float4 o0 : SV_Target0)
{
  float4 zpos4 = InjectedDepthTexture.Load((int3) injectedScreenPos.xyz);
  float zpos = zpos4.x - 1;
  float wpos = 1.0 / zpos;
  float4 r0,r1,r2,r3,r4,r5,r6,r7,r8;
  uint4 bitmask;
  r0.xy = float2(0.000000e+000,0.000000e+000);
  r1.xy = v1.xy / v1.ww;
  r1.xy = r1.xy * g_ViewportScaleOffset.xy + g_ViewportScaleOffset.zw;
  r2.xyzw = DepthSurface.SampleLevel(DepthSurface_s, r1.xy, 0.000000000e+000).xyzw;
  r0.w = g_ReverseProjectionParams.z + r2.x;
  r0.w = g_ReverseProjectionParams.w / r0.w;
  r2.xyz = v2.xyz * r0.www + v3.xyz;

//Mike_ar69 - Stereo Shift
float4 stereoParams = StereoParams.Load(0);
float4 r21, r22;
r21.xyz = r2.xyz;
r21.w = 1.000000000e+000;
r22.x = dot(r21.xyzw, g_WorldToView._m00_m10_m20_m30);
r22.y = dot(r21.xyzw, g_WorldToView._m01_m11_m21_m31);
r22.z = dot(r21.xyzw, g_WorldToView._m02_m12_m22_m32);
r22.w = dot(r21.xyzw, g_WorldToView._m03_m13_m23_m33);
r22.x +=  stereoParams.x * (-r22.z - stereoParams.y ) * g_ReverseProjectionParams.x;
r21.x = dot(r22.xyzw, g_ViewToWorld._m00_m10_m20_m30);
r21.y = dot(r22.xyzw, g_ViewToWorld._m01_m11_m21_m31);
r21.z = dot(r22.xyzw, g_ViewToWorld._m02_m12_m22_m32);
r2.xyz = r21.xyz;


  r1.zw = r2.xy * g_WorldLightMapUVParameters.zw + g_WorldLightMapUVParameters.xy;
  r3.xyzw = g_WorldLightMapDirectSampler.SampleLevel(g_WorldLightMapDirectSampler_s, r1.zw, 0.000000000e+000).xyzw;
  r1.zw = r3.xy * float2(2.000000e+000,2.000000e+000) + float2(-1.000000e+000,-1.000000e+000);
  r4.xy = g_WorldLightMapParameters1.ww * r1.zw;
  r4.z = -r2.z;
  r2.xyz = g_EyePosition.xyz + -r2.xyz;
  r0.z = r3.z * g_WorldLightMapParameters1.x + g_WorldLightMapParameters1.y;
  r0.xyz = r4.xyz + r0.xyz;
  r0.w = dot(r0.xyz, r0.xyz);
  r0.w = rsqrt(r0.w);
  r0.xyw = r0.xyz * r0.www;
  r0.z = g_WorldLightMapParameters1.z * abs(r0.z);
  r0.z = -r0.z * r0.z + 1.000000000e+000;
  r0.z = saturate(r3.w * r0.z);
  r3.xyz = g_WorldLightMapParameters3.xyz * r0.zzz;
  r0.z = dot(r2.xyz, r2.xyz);
  r0.z = rsqrt(r0.z);
  r4.xyz = r2.xyz * r0.zzz + r0.xyw;
  r2.xyz = r2.xyz * r0.zzz + g_DeferredDirectLight.m_Direction.xyz;
  r0.z = dot(r4.xyz, r4.xyz);
  r0.z = rsqrt(r0.z);
  r4.xyz = r4.xyz * r0.zzz;
  r5.xyzw = g_Normals_NG.Sample(g_PointClampSampler_NG_s, r1.xy).xyzw;
  r5.xyz = r5.xyz * float3(2.000000e+000,2.000000e+000,2.000000e+000) + float3(-1.000000e+000,-1.000000e+000,-1.000000e+000);
  r0.z = r5.w + r5.w;
  r1.z = dot(r5.xyz, r5.xyz);
  r1.z = rsqrt(r1.z);
  r6.xyz = r5.xyz * r1.zzz;
  r1.z = dot(g_DeferredDirectLight.m_Direction.xyz, r5.xyz);
  r1.w = saturate(dot(r6.xyz, r4.xyz));
  r1.w = log2(r1.w);
  r4.xyzw = g_LightingAccumulation_NG.Sample(g_PointClampSampler_NG_s, r1.xy).xyzw;
  r2.w = 8.479969025e+000 * r4.w;
  r2.w = exp2(r2.w);
  r1.w = r2.w * r1.w;
  r1.w = exp2(r1.w);
  r0.x = saturate(dot(r6.xyz, r0.xyw));
  r0.y = r0.x * r1.w;
  r5.xyz = r0.xxx * r3.xyz;
  r0.x = r2.w * 2.000000000e+000 + -2.000000000e+000;
  r0.x = 1.250000000e-001 * r0.x;
  r0.y = r0.y * r0.x;
  r3.xyz = r0.yyy * r3.xyz;
  r3.xyz = float3(6.000000e-002,6.000000e-002,6.000000e-002) * r3.xyz;
  r7.xyzw = g_Albedo_NG.Sample(g_PointClampSampler_NG_s, r1.xy).xyzw;
  r8.xyzw = g_AmbientTexture.Sample(g_AmbientTexture_s, r1.xy).xyzw;
  r1.xyw = r5.xyz * r7.xyz + r3.xyz;
  r1.xyw = r4.xyz + r1.xyw;
  o0.w = r4.w;
  r0.y = 3.000000119e-001 + -r1.z;
  r1.z = saturate(r1.z);
  r0.y = r0.z * r0.y;
  r0.y = max(r0.y, 0.000000000e+000);
  r0.y = r1.z + r0.y;
  r3.xyz = g_DeferredDirectLight.m_Color.xyz * r7.xyz;
  r4.xyz = saturate(r8.xyz * r7.xyz);
  r3.xyz = r3.xyz * r8.www;
  r0.yzw = r3.xyz * r0.yyy + r4.xyz;
  r3.x = dot(r2.xyz, r2.xyz);
  r3.x = rsqrt(r3.x);
  r2.xyz = r3.xxx * r2.xyz;
  r2.x = saturate(dot(r6.xyz, r2.xyz));
  r2.x = log2(r2.x);
  r2.x = r2.w * r2.x;
  r2.x = exp2(r2.x);
  r1.z = r2.x * r1.z;
  r0.x = r1.z * r0.x;
  r2.xyz = g_DeferredDirectLight.m_Color.xyz * r0.xxx;
  r2.xyz = r2.xyz * r8.www;
  r0.xyz = r2.xyz * float3(6.000000e-002,6.000000e-002,6.000000e-002) + r0.yzw;
  o0.xyz = r0.xyz + r1.xyw;
  return;
}
