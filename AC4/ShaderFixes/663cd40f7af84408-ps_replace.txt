void main(
  float4 injectedScreenPos : SV_Position,
  float4 v0 : SV_Position0,
  float4 v1 : TEXCOORD0,
  float3 v2 : TEXCOORD1,
  float3 v3 : TEXCOORD2,
  uint v4 : SV_SampleIndex0,
  out float4 o0 : SV_Target0)
{
  float4 zpos4 = InjectedDepthTexture.Load((int3) injectedScreenPos.xyz);
  float zpos = zpos4.x - 1;
  float wpos = 1.0 / zpos;
  float4 r0,r1,r2,r3,r4,r5,r6;
  uint4 bitmask;
  r0.zw = float2(0.000000e+000,0.000000e+000);
  g_TextureDepthDX11_MSAA.GetDimensions(r1.x, r1.y, bitmask.x);
  r1.xy = r1.xy;
  r1.zw = v1.xy / v1.ww;
  r1.zw = r1.zw * g_ViewportScaleOffset.xy + g_ViewportScaleOffset.zw;
  r1.xy = r1.zw * r1.xy;
  r0.xy = (int2)r1.xy;
  r0.x = g_TextureDepthDX11_MSAA.Load(r0.xyz,v4.x).x;
  r0.x = g_ReverseProjectionParams.z + r0.x;
  r0.x = g_ReverseProjectionParams.w / r0.x;
  r0.xyz = v2.xyz * r0.xxx + v3.xyz;

//Mike_ar69 - Stereo Shift
float4 stereoParams = StereoParams.Load(0);
float4 r21, r22;
r21.xyz = r0.xyz;
r21.w = 1.000000000e+000;
r22.x = dot(r21.xyzw, g_WorldToView._m00_m10_m20_m30);
r22.y = dot(r21.xyzw, g_WorldToView._m01_m11_m21_m31);
r22.z = dot(r21.xyzw, g_WorldToView._m02_m12_m22_m32);
r22.w = dot(r21.xyzw, g_WorldToView._m03_m13_m23_m33);
r22.x +=  stereoParams.x * (-r22.z - stereoParams.y ) * g_ReverseProjectionParams.x;
r21.x = dot(r22.xyzw, g_ViewToWorld._m00_m10_m20_m30);
r21.y = dot(r22.xyzw, g_ViewToWorld._m01_m11_m21_m31);
r21.z = dot(r22.xyzw, g_ViewToWorld._m02_m12_m22_m32);
r0.xyz = r21.xyz;


  g_TextureNormalDX11_MSAA.GetDimensions(r1.x, r1.y, bitmask.x);
  r1.xy = r1.xy;
  r1.xy = r1.zw * r1.xy;
  r2.xy = (int2)r1.xy;
  r2.zw = float2(0.000000e+000,0.000000e+000);
  r2.xyzw = g_TextureNormalDX11_MSAA.Load(r2.xyz,v4.x).xyzw;
  r2.xyz = r2.xyz * float3(2.000000e+000,2.000000e+000,2.000000e+000) + float3(-1.000000e+000,-1.000000e+000,-1.000000e+000);
  r0.w = r2.w + r2.w;
  r1.x = dot(r2.xyz, r2.xyz);
  r1.x = rsqrt(r1.x);
  r3.xyz = r2.xyz * r1.xxx;
  r1.x = dot(g_DeferredDirectLight.m_Direction.xyz, r2.xyz);
  r2.xy = r3.xy * g_WorldLightMapParameters2.xx + r0.xy;
  r2.xy = r2.xy * g_WorldLightMapUVParameters.zw + g_WorldLightMapUVParameters.xy;
  r2.xyzw = g_WorldLightmapIndirectSampler.SampleLevel(g_WorldLightmapIndirectSampler_s, r2.xy, 0.000000000e+000).xyzw;
  r1.y = r2.w * g_WorldLightMapParameters1.x + g_WorldLightMapParameters1.y;
  r1.y = r1.y + -r0.z;
  r1.y = r3.z * g_WorldLightMapParameters2.y + r1.y;
  r1.y = g_WorldLightMapParameters1.z * abs(r1.y);
  r1.y = -r1.y * r1.y + 1.000000000e+000;
  r1.y = max(r1.y, 0.000000000e+000);
  r1.y = g_WorldLightMapParameters3.w * r1.y;
  r2.xyz = r2.xyz * r1.yyy;
  r4.z = -r0.z;
  r5.xy = r0.xy * g_WorldLightMapUVParameters.zw + g_WorldLightMapUVParameters.xy;
  r0.xyz = g_EyePosition.xyz + -r0.xyz;
  r5.xyzw = g_WorldLightMapDirectSampler.SampleLevel(g_WorldLightMapDirectSampler_s, r5.xy, 0.000000000e+000).xyzw;
  r5.xy = r5.xy * float2(2.000000e+000,2.000000e+000) + float2(-1.000000e+000,-1.000000e+000);
  r4.xy = g_WorldLightMapParameters1.ww * r5.xy;
  r5.z = r5.z * g_WorldLightMapParameters1.x + g_WorldLightMapParameters1.y;
  r5.xy = float2(0.000000e+000,0.000000e+000);
  r4.xyz = r5.xyz + r4.xyz;
  r1.y = g_WorldLightMapParameters1.z * abs(r4.z);
  r1.y = -r1.y * r1.y + 1.000000000e+000;
  r1.y = saturate(r5.w * r1.y);
  r5.xyz = g_WorldLightMapParameters3.xyz * r1.yyy;
  r1.y = dot(r4.xyz, r4.xyz);
  r1.y = rsqrt(r1.y);
  r4.xyz = r4.xyz * r1.yyy;
  r1.y = saturate(dot(r3.xyz, r4.xyz));
  r2.xyz = r1.yyy * r5.xyz + r2.xyz;
  r2.w = dot(r0.xyz, r0.xyz);
  r2.w = rsqrt(r2.w);
  r4.xyz = r0.xyz * r2.www + r4.xyz;
  r0.xyz = r0.xyz * r2.www + g_DeferredDirectLight.m_Direction.xyz;
  r2.w = dot(r4.xyz, r4.xyz);
  r2.w = rsqrt(r2.w);
  r4.xyz = r4.xyz * r2.www;
  r2.w = saturate(dot(r3.xyz, r4.xyz));
  r2.w = log2(r2.w);
  g_TextureLightingAccumulationDX11_MSAA.GetDimensions(r4.x, r4.y, bitmask.x);
  r4.xy = r4.xy;
  r4.xy = r4.xy * r1.zw;
  r4.xy = (int2)r4.xy;
  r4.zw = float2(0.000000e+000,0.000000e+000);
  r4.xyzw = g_TextureLightingAccumulationDX11_MSAA.Load(r4.xyz,v4.x).xyzw;
  r3.w = 8.479969025e+000 * r4.w;
  r3.w = exp2(r3.w);
  r2.w = r3.w * r2.w;
  r2.w = exp2(r2.w);
  r1.y = r2.w * r1.y;
  r2.w = r3.w * 2.000000000e+000 + -2.000000000e+000;
  r2.w = 1.250000000e-001 * r2.w;
  r1.y = r2.w * r1.y;
  r5.xyz = r1.yyy * r5.xyz;
  r5.xyz = float3(6.000000e-002,6.000000e-002,6.000000e-002) * r5.xyz;
  g_TextureAlbedoDX11_MSAA.GetDimensions(r6.x, r6.y, bitmask.x);
  r6.xy = r6.xy;
  r6.xy = r6.xy * r1.zw;
  r6.xy = (int2)r6.xy;
  r6.zw = float2(0.000000e+000,0.000000e+000);
  r6.xyz = g_TextureAlbedoDX11_MSAA.Load(r6.xyz,v4.x).xyz;
  r2.xyz = r2.xyz * r6.xyz + r5.xyz;
  r2.xyz = r4.xyz + r2.xyz;
  o0.w = r4.w;
  r1.y = 3.000000119e-001 + -r1.x;
  r1.x = saturate(r1.x);
  r0.w = r1.y * r0.w;
  r0.w = max(r0.w, 0.000000000e+000);
  r0.w = r1.x + r0.w;
  r4.xyz = g_DeferredDirectLight.m_Color.xyz * r6.xyz;
  g_AmbientTextureDX11_MSAA.GetDimensions(r5.x, r5.y, bitmask.x);
  r5.xy = r5.xy;
  r1.yz = r5.xy * r1.zw;
  r5.xy = (int2)r1.yz;
  r5.zw = float2(0.000000e+000,0.000000e+000);
  r5.xyzw = g_AmbientTextureDX11_MSAA.Load(r5.xyz,v4.x).xyzw;
  r1.yzw = r5.www * r4.xyz;
  r4.xyz = saturate(r5.xyz * r6.xyz);
  r1.yzw = r1.yzw * r0.www + r4.xyz;
  r0.w = dot(r0.xyz, r0.xyz);
  r0.w = rsqrt(r0.w);
  r0.xyz = r0.xyz * r0.www;
  r0.x = saturate(dot(r3.xyz, r0.xyz));
  r0.x = log2(r0.x);
  r0.x = r3.w * r0.x;
  r0.x = exp2(r0.x);
  r0.x = r1.x * r0.x;
  r0.x = r0.x * r2.w;
  r0.xyz = g_DeferredDirectLight.m_Color.xyz * r0.xxx;
  r0.xyz = r0.xyz * r5.www;
  r0.xyz = r0.xyz * float3(6.000000e-002,6.000000e-002,6.000000e-002) + r1.yzw;
  o0.xyz = r0.xyz + r2.xyz;
  return;
}
