
cbuffer vc2 : register(b2)
{
  float4x4 projTM : packoffset(c0);
  float4 eyePos : packoffset(c4);
  float4 Fog_dist : packoffset(c10);
  float4x4 IR_World2View : packoffset(c22);
  float Reflection_Plane_Height : packoffset(c30);
}

Texture2D<float4> StereoParams : register(t125);

void main(
  float3 v0 : POSITION0,
  int2 v1 : TEXCOORD0,
  float3 v2 : NORMAL0,
  out float4 o0 : SV_Position0,
  out float4 o1 : TEXCOORD0,
  out float3 o2 : TEXCOORD1)
{
  float4 r0,r1;
  uint4 bitmask;
  o0.x = dot(v0.xyz, projTM._m00_m10_m20);
  o0.y = dot(v0.xyz, projTM._m01_m11_m21);
  o0.z = dot(v0.xyz, projTM._m02_m12_m22);
  o0.w = dot(v0.xyz, projTM._m03_m13_m23);
  r0.xyz = -eyePos.xyz + v0.xyz;
  r0.w = Fog_dist.x * r0.y;
  r1.x = -1.442695022e+000 * r0.w;
  r1.x = exp2(r1.x);
  r1.x = 1.000000000e+000 + -r1.x;
  r0.w = r1.x / r0.w;
  r1.x = 9.999999776e-003 < abs(r0.y);
  r0.x = dot(r0.xyz, r0.xyz);
  r0.x = sqrt(r0.x);
  r0.x = Fog_dist.y * r0.x;
  r0.y = r1.x ? r0.w : 1.000000000e+000;
  r0.x = r0.y * r0.x + -Fog_dist.w;
  r0.x = exp2(-r0.x);
  r0.x = min(r0.x, 1.000000000e+000);
  o1.w = 1.000000000e+000 + -r0.x;
  r0.xyz = v2.xyz * float3(2.000000e+000,2.000000e+000,2.000000e+000) + float3(-1.000000e+000,-1.000000e+000,-1.000000e+000);
  o1.x = dot(r0.xyz, IR_World2View._m00_m10_m20);
  o1.y = dot(r0.xyz, IR_World2View._m01_m11_m21);
  o1.z = dot(r0.xyz, IR_World2View._m02_m12_m22);
  r0.xy = v1.xy;
  o2.xy = float2(9.765625e-004,9.765625e-004) * r0.xy;
  o2.z = -Reflection_Plane_Height + v0.y;
  return;
}
